97  void update(double deltaTime) {
98  //move camera and target in viewing direction
99  if(mFront || mBack || mLeft || mRight) {
100  Vector movement;
101  front = (target - position).normalize();
102  if (deltaTime < 0.000007) deltaTime = 0.000007;
103  movement = front * (CAMERA_MOVEMENTSPEED * deltaTime);
104  if(mFront || mBack) {
105  if(mFront) {
106  position = position + movement;
107  target = target + movement;
108  }
109  else {
110  position = position - movement;
111  target = target - movement;
112  }
113 
114  // See if we are through the ground / height limit...
115  if(position.y < CAMERA_VERTICAL_LIMIT ) {
116  Vector push = Vector(0.0, CAMERA_VERTICAL_LIMIT - position.y, 0.0);
117  position = position + push;
118  target = target + push;
119  }
120  }
121  if(mLeft || mRight) {
122  movement = (front % up) * (CAMERA_MOVEMENTSPEED * deltaTime);
123 
124  if(mLeft) {
125  position = position - movement;
126  target = target - movement;
127  }
128  else {
129  position = position + movement;
130  target = target + movement;
131  }
132  }
133  }
134

138  position = position - pull;
