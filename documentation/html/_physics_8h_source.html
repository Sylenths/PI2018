344  static Vector getClosestPointOnLineSegment(Vector searchOrigin, Vector a, Vector b) {
345 
346  // Determine length of A-->Point vector.
347  Vector c = searchOrigin - a;
348 
349  // Get a normalised vector that is co-linear with AB for projection.
350  Vector V = (b - a).normalize();
351 
352  double j = (b.x - a.x), k = (b.y - a.y), l = (b.z - a.z);
353  double d = j*j + k*k + l*l; // Do not square root this.
354 
355  double projectionLength = V * c;
356 
357  // See if projected point is not on the segment and, if so, return the appropriate end of the segment.
358 
359  if (projectionLength < 0) return a;
360 
361  if ((projectionLength * projectionLength) > d) return b; // Avoid computing the square root of d by squaring t, less processor intensive.
362 
363  // Return projected point on segment
364 
365  V.normalize();
366  V = V * projectionLength;
367 
368  return a + V;
369 
370  }

411  static PhysicsData::CollisionData collideVectorOnSphere(Vector vectorOrigin, Vector vector, Vector sphereOrigin, double sphereRadius){
412  Vector normalizedVector(vector.getNormalized());
413 
414  Vector Q = sphereOrigin - vectorOrigin; // Vector from vectorOrigin to sphereOrigin
415 
416  double c = Q.x * Q.x + Q.y * Q.y + Q.z * Q.z; //distance squared between vectorOrigin and sphereOrigin;
417 
418  double v = Q * normalizedVector; // Project Q on vector to collide
419 
420  double d = sphereRadius*sphereRadius - (c - v*v);
421 
422  // Failure to intersect, return false.
423  if (d < 0.0) {
424  return {false};
425  }
426 
427 
428  double distanceToIntersect = (v - sqrt(d));
429 
430  return {true, vectorOrigin + (normalizedVector * distanceToIntersect), distanceToIntersect / vector.getNorm()};
431  }
