77  static bool isPointWithinTri(Vector point, PhysicsData::Triangle tri, Vector normal){
78  PhysicsData::FlatTriangle flatTri;
79  PhysicsData::Vec2D flatVec;
80 
81  if((normal.x > normal.y) && (normal.x > normal.z)){
82  //remove x from tri
83  flatTri = {{tri.p1.y, tri.p1.z}, {tri.p2.y, tri.p2.z}, {tri.p3.y, tri.p3.z}};
84  flatVec = {point.y, point.z};
85  }
86  else if(normal.y > normal.z){
87  //remove y from tri
88  flatTri = {{tri.p1.x, tri.p1.z}, {tri.p2.x, tri.p2.z}, {tri.p3.x, tri.p3.z}};
89  flatVec = {point.x, point.z};
90  }
91  else{
92  //remove z from tri
93  flatTri = {{tri.p1.x, tri.p1.y}, {tri.p2.x, tri.p2.y}, {tri.p3.x, tri.p3.y}};
94  flatVec = {point.x, point.y};
95  }
96 
97  bool b1 = signbit(det22(flatVec.x, flatVec.y, flatTri.p1.x, flatTri.p1.y));
98  bool b2 = signbit(det22(flatVec.x, flatVec.y, flatTri.p2.x, flatTri.p2.y));
99  bool b3 = signbit(det22(flatVec.x, flatVec.y, flatTri.p3.x, flatTri.p3.y));
100 
101  if(b1 == b2 && b1 == b3){
102  //Collision inside triangle.
103  return true;
104  }
105  else{
106  //No collision inside triangle.
107  return false;
108  }

315  static Vector getClosestPointOnLineSegment(Vector searchOrigin, Vector a, Vector b) {
316 
317  // Determine length of A-->Point vector.
318  Vector c = searchOrigin - a;
319 
320  // Get a normalised vector that is co-linear with AB for projection.
321  Vector V = (b - a).normalize();
322 
323  double j = (b.x - a.x), k = (b.y - a.y), l = (b.z - a.z);
324  double d = j*j + k*k + l*l; // Do not square root this.
325 
326  double projectionLength = V * c;
327 
328  // See if projected point is not on the segment and, if so, return the appropriate end of the segment.
329 
330  if (projectionLength < 0) return a;
331 
332  if ((projectionLength * projectionLength) > d) return b; // Avoid computing the square root of d by squaring t, less processor intensive.
333 
334  // Return projected point on segment
335 
336  V.normalize();
337  V = V * projectionLength;
338 
339  return a + V;
340 
341  }

363  static PhysicsData::CollisionData collideVectorOnSphere(Vector vectorOrigin, Vector vector, Vector sphereOrigin, double sphereRadius){
364  Vector normalizedVector(vector.getNormalized());
365 
366  Vector Q = sphereOrigin - vectorOrigin;
367 
368  double c = Q.getNorm();
369 
370  double v = Q * vector.getNormalized();
371 
372  double d = sphereRadius*sphereRadius - (c*c - v*v);
373 
374  // Failure to intersect, return false.
375  if (d < 0.0) {
376  return {false};
377  }
378 
379 
380  double distanceToIntersect = (v - sqrt(d));
381 
382  return {true, vectorOrigin + (normalizedVector * distanceToIntersect), distanceToIntersect / vector.getNorm()};
383  }
