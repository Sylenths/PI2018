56  static Physics::CollisionData collideVectorOnTri(Vector segmentOrigin, Vector segment, Physics::Triangle tri, Vector normal){
57  normal.normalize();
58 
59  double dotp = normal * segment;
60 
61  if((PHYSICS_EPSILON > dotp) && (dotp > -PHYSICS_EPSILON))
62  return {false};
63 
64  double ratio = ((normal * tri.p1) / (normal * segment));
65 
66  if(ratio > 1)
67  return {false};
68 
69  Vector intersect(segmentOrigin + (segment * ratio));
70 
71  Physics::FlatTriangle flatTri;
72  Physics::Vec2D flatVec;
73 
74  if((normal.x > normal.y) && (normal.x > normal.z)){
75  //remove x from tri
76  flatTri = {{tri.p1.y, tri.p1.z}, {tri.p2.y, tri.p2.z}, {tri.p3.y, tri.p3.z}};
77  flatVec = {intersect.y, intersect.z};
78  }
79  else if(normal.y > normal.z){
80  //remove y from tri
81  flatTri = {{tri.p1.x, tri.p1.z}, {tri.p2.x, tri.p2.z}, {tri.p3.x, tri.p3.z}};
82  flatVec = {intersect.x, intersect.z};
83  }
84  else{
85  //remove z from tri
86  flatTri = {{tri.p1.x, tri.p1.y}, {tri.p2.x, tri.p2.y}, {tri.p3.x, tri.p3.y}};
87  flatVec = {intersect.x, intersect.y};
88  }
89 
90  bool b1 = signbit(det22(flatVec.x, flatVec.y, flatTri.p1.x, flatTri.p1.y));
91  bool b2 = signbit(det22(flatVec.x, flatVec.y, flatTri.p2.x, flatTri.p2.y));
92  bool b3 = signbit(det22(flatVec.x, flatVec.y, flatTri.p3.x, flatTri.p3.y));
93 
94  if(b1 == b2 && b1 == b3){
95  //Collision inside triangle.
96  return {true, intersect, ratio};
97  }
98  else{
99  //No collision inside triangle.
100  return {false};
101  }

103  }
