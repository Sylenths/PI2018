105  void loadArbitraryRotation(Vector p, Vector a, double angle){
106  a.normalize();
107 
108  double u2 = a.x*a.x;
109  double v2 = a.y*a.y;
110  double w2 = a.z*a.z;
111  double c = std::cos(degtorad(angle));
112  double _1_c = 1 - c;
113  double s = std::sin(degtorad(angle));
114 
115  /* Neat trick : to invert the positive rotation direction (aka. "default rotation direction"),
116  * invert the plus or minus between the two terms of everything in the 3x3 part of the matrix that isn't on the
117  * top-left bottom-right diagonal; this amounts to a matrix transposition in this specific case.
118  */
119  m11 = u2 + (v2 + w2) * c;
120  m12 = a.x*a.y * _1_c + a.z*s;
121  m13 = a.x*a.z * _1_c - a.y*s;
122  m14 = (p.x*(v2 + w2) + a.x*(p.y*a.y + p.z*a.z)) * _1_c + (p.y*a.z - p.z*a.y)*s;
123 
124  m21 = a.x*a.y * _1_c - a.z*s;
125  m22 = v2 + (u2 + w2) * c;
126  m23 = a.y*a.z * _1_c + a.x*s;
127  m24 = (p.y*(u2 + w2) + a.y*(p.x*a.x + p.z*a.z)) * _1_c + (p.z*a.x - p.x*a.z)*s;
128 
129  m31 = a.x*a.z * _1_c + a.y*s;
130  m32 = a.y*a.z * _1_c - a.x*s;
131  m33 = w2 + (u2 + v2) * c;
132  m34 = (p.z*(u2 + v2) + a.z*(p.x*a.x + p.y*a.y)) * _1_c + (p.x*a.y - p.y*a.x)*s;
133 
134  m41 = m42 = m43 = 0.;
135  m44 = 1.;
136 
137  }
